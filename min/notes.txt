13. Example Core Validation Code

Simplified:

import aiohttp
import asyncio

async def validate_url(session, url):

    try:
        async with session.head(url, timeout=10) as r:
            return (url, r.status)

    except:
        try:
            async with session.get(url, timeout=10) as r:
                return (url, r.status)
        except Exception as e:
            return (url, str(e))
